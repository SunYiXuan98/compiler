int mod(int a, int b)
{
	while(a >= b)
	{	
		a = a - b;
	}
	return(a);
}

void factorize(int n)
{
	int p;
	int flag;
	
		while(mod(n, p) == 0)
		{
			if(flag)
			{
				printf(p);
				flag = 0;
			}
			else{
				printf("*");
				printf(p);
			} 
			n = n / p;
		}

}
